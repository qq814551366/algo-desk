# 实例
``` c++

 vector<int> sortedSquares(vector<int>& nums) {
        int n = nums.size();
        vector<int> res (n);
        for(int i =0,j = n-1,p =j;p>=0;){
            int start = nums[i] * nums[i];
            int end = nums[j] * nums[j];
            if(start>end){
                res[p] = start;
                i ++;
            }else{
                res[p] = end;
                j--;
            }
            p--;
        }
        return res;
    }
```
